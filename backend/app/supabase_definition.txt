
## Supabase Table Definitions

create table
  public.custom_model (
    id bigint generated by default as identity,
    created_at timestamp with time zone null default now(),
    user_id uuid null,
    voice_id text null,
    samples_used jsonb null,
    constraint custom_models_pkey primary key (id),
    constraint custom_model_user_id_fkey foreign key (user_id) references auth.users (id)
  ) tablespace pg_default;

  create table
  public.custom_model (
    id bigint generated by default as identity,
    created_at timestamp with time zone null default now(),
    user_id uuid null,
    voice_id text null,
    samples_used jsonb null,
    constraint custom_models_pkey primary key (id),
    constraint custom_model_user_id_fkey foreign key (user_id) references auth.users (id)
  ) tablespace pg_default;

  create table
  public.project (
    id bigint generated by default as identity,
    created_at timestamp with time zone null default now(),
    name text null,
    user_id uuid null,
    files bigint null,
    constraint project_pkey primary key (id),
    constraint project_files_fkey foreign key (files) references file (id),
    constraint project_user_id_fkey foreign key (user_id) references auth.users (id)
  ) tablespace pg_default;

  create table
  public.synthesis (
    id bigint generated by default as identity,
    created_at timestamp with time zone null default now(),
    is_clone boolean null,
    user_id uuid null,
    source_file bigint null,
    request_content text null,
    project_id bigint null,
    synthesized_file bigint null,
    voice_id text null,
    constraint synthesis_pkey primary key (id),
    constraint synthesis_project_id_fkey foreign key (project_id) references project (id),
    constraint synthesis_source_file_fkey foreign key (source_file) references file (id),
    constraint synthesis_user_id_fkey foreign key (user_id) references auth.users (id)
  ) tablespace pg_default;

  create table
  public.user (
    aud text null,
    role text null,
    email character varying null,
    app_metadata jsonb null,
    user_metadata jsonb null,
    user_id uuid not null,
    constraint user_pkey primary key (user_id),
    constraint user_user_id_fkey foreign key (user_id) references auth.users (id)
  ) tablespace pg_default;

  create table
  public.voice_samples (
    id bigint generated by default as identity,
    file_path text null,
    created_at timestamp with time zone null default now(),
    user_id uuid null,
    audio_metadata jsonb null,
    constraint voice_samples_pkey primary key (id),
    constraint voice_samples_user_id_fkey foreign key (user_id) references auth.users (id)
  ) tablespace pg_default;
